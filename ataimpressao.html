// ===== BUSCAR DADOS DA ATA DO CIPA MANAGER ===== async function buscarAtaCipaManager() { const select = document.getElementById('selectBuscarAta'); const idAta = select ? select.value.trim() : ''; if (!idAta) { showAlert('❌ Selecione uma ATA na lista!'); return; } const btn = document.getElementById('btnBuscarAta'); const textoOriginal = btn.textContent; btn.textContent = '⏳ Buscando...'; btn.disabled = true; try { const url = `${GOOGLE_API_URL_CIPA_MANAGER}?acao=carregar&aba=atas&id=${encodeURIComponent(idAta)}`; const response = await fetch(url); const resultado = await response.json(); if (resultado.status !== 'sucesso' || !resultado.dados) { showAlert('❌ ATA não encontrada no CIPA Manager! Verifique a seleção.'); btn.textContent = textoOriginal; btn.disabled = false; return; } // CORREÇÃO AQUI: verifica se já é objeto ou string let dados; if (typeof resultado.dados === 'string') { dados = JSON.parse(resultado.dados); } else { dados = resultado.dados; } preencherCamposAta(dados); showAlert('✅ ATA carregada com sucesso! Campos preenchidos.'); } catch (error) { console.error('Erro ao buscar ata:', error); showAlert('❌ Erro ao buscar ata: ' + error.message); } finally { btn.textContent = textoOriginal; btn.disabled = false; } }
