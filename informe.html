// ===== QUANDO SELECIONAR UM SETOR NO MODAL, ADICIONA OS EMAILS ===== const selectSetorEnvio = document.getElementById('selectSetorEnvio'); if (selectSetorEnvio) { selectSetorEnvio.addEventListener('change', function() { const email = this.value; const nome = this.options[this.selectedIndex]?.dataset?.nome || ''; if (!email || !email.trim()) { return; } // Pega os emails do setor (pode ter vários separados por vírgula) const novosEmails = email.split(',').map(e => e.trim()).filter(e => e); // ADICIONA os emails à lista (sem duplicatas) let adicionados = 0; novosEmails.forEach(e => { if (!destinatarios.some(d => d.email === e)) { destinatarios.push({ email: e, nome: nome }); adicionados++; } }); if (adicionados > 0) { showAlert(`✅ ${adicionados} email(s) adicionado(s) à lista!`, 'success'); } else { showAlert('ℹ️ Todos os emails já estavam na lista!', 'info'); } atualizarListaDestinatarios(); // Limpa o select para permitir selecionar o mesmo setor novamente this.value = ''; }); }
